```cpp
// sqlite_manager.h
#ifndef SQLite_MANAGER_H

#include <string>
#include <vector>
#include <sqlite3.h>

struct ColumnInfo {
    std::string name;
    std::string dataType;
    bool notNull;
    bool primaryKey;
    std::string defaultValue;

    std::string name;
    std::string type;  // "table" or "view"
    int rowCount;
    std::vector<ColumnInfo> columns;
};

struct IndexInfo {
    std::string name;
    std::string tableName;
    bool unique;
    std::vector<std::string> columns;

struct QueryResult {
    bool success;
    std::vector<std::string> columns;
    std::vector<std::vector<std::string>> rows;
    int affectedRows;
    long long executionTime;
    std::string error;
};

class SqliteManager {
public:
    SqliteManager();
    ~SqliteManager();

    // 数据库操作
    bool openDatabase(const std::string& path);
    bool closeDatabase();
    bool isOpen() const { return m_db != nullptr; }
    std::string getDatabasePath() const { return m_dbPath; }

    std::vector<std::string> getTableNames();
    std::vector<std::string> getViewNames();
    std::vector<TableInfo> getTables();
    std::vector<IndexInfo> getIndexes(const std::string& tableName = "");
    std::vector<ColumnInfo> getColumns(const std::string& tableName);

    QueryResult rawQuery(const std::string& sql);
    std::string explain(const std::string& sql);

    // 事务
    bool beginTransaction();
    bool commit();
    bool rollback();

    // 工具
    std::string getTableDDL(const std::string& tableName);
    int getRowCount(const std::string& tableName);

    bool m_inTransaction;

    std::string getLastError();

#endif // SQLite_MANAGER_H
```

```cpp
// sqlite_manager.cpp
#include "sqlite_manager.h"
#include <chrono>

SqliteManager::SqliteManager() : m_db(nullptr), m_inTransaction(false) {}

SqliteManager::~SqliteManager() {
    if (m_db) {
        sqlite3_close(m_db);
        m_db = nullptr;
    }
}

bool SqliteManager::openDatabase(const std::string& path) {
    if (m_db) {
        closeDatabase();
    }

    int result = sqlite3_open(path.c_str(), &m_db);
    if (result != SQLITE_OK) {
        return false;
    }
    m_dbPath = path;
    return true;
}

bool SqliteManager::closeDatabase() {
    if (m_db) {
        if (m_inTransaction) {
            rollback();
        }
        sqlite3_close(m_db);
        m_db = nullptr;
        m_dbPath = "";
        return true;
    }
    return false;
}

std::vector<std::string> SqliteManager::getTableNames() {
    std::vector<std::string> tables;
    if (!m_db) return tables;

    const char* sql = "SELECT name FROM sqlite_master WHERE type='table' AND name NOT LIKE 'sqlite_%' ORDER BY name";
    sqlite3_stmt* stmt;

    if (sqlite3_prepare_v2(m_db, sql, -1, &stmt, nullptr) == SQLITE_OK) {
        while (sqlite3_step(stmt) == SQLITE_ROW) {
            const char* name = (const char*)sqlite3_column_text(stmt, 0);
            tables.push_back(name);
        }
    }
    sqlite3_finalize(stmt);
    return tables;
}

std::vector<ColumnInfo> SqliteManager::getColumns(const std::string& tableName) {
    std::vector<ColumnInfo> columns;
    if (!m_db) return columns;

    std::string sql = "PRAGMA table_info('" + tableName + "')";
    sqlite3_stmt* stmt;

    if (sqlite3_prepare_v2(m_db, sql.c_str(), -1, &stmt, nullptr) == SQLITE_OK) {
        while (sqlite3_step(stmt) == SQLITE_ROW) {
            ColumnInfo col;
            col.name = (const char*)sqlite3_column_text(stmt, 1);
            col.dataType = (const char*)sqlite3_column_text(stmt, 2);
            col.notNull = sqlite3_column_int(stmt, 3) != 0;
            col.defaultValue = sqlite3_column_type(stmt, 4) == SQLITE_NULL ? "" :
                              std::string((const char*)sqlite3_column_text(stmt, 4));
            col.primaryKey = sqlite3_column_int(stmt, 5) != 0;
            columns.push_back(col);
        }
    }
    sqlite3_finalize(stmt);
    return columns;
}

QueryResult SqliteManager::execute(const std::string& sql, const std::vector<std::string>& args) {
    auto start = std::chrono::high_resolution_clock::now();

    QueryResult result;
    result.success = true;
    result.affectedRows = 0;

    if (!m_db) {
        result.success = false;
        result.error = "Database not opened";
        return result;
    }

    sqlite3_stmt* stmt;
    int prepareResult = sqlite3_prepare_v2(m_db, sql.c_str(), -1, &stmt, nullptr);

    if (prepareResult != SQLITE_OK) {
        result.success = false;
        result.error = sqlite3_errmsg(m_db);
        return result;
    }

    // Bind arguments
    for (size_t i = 0; i < args.size() && i < (size_t)sqlite3_bind_parameter_count(stmt); ++i) {
        sqlite3_bind_text(stmt, i + 1, args[i].c_str(), -1, SQLITE_TRANSIENT);
    }

    // Execute and collect results
    bool isQuery = sqlite3_column_count(stmt) > 0;

    while (true) {
        int stepResult = sqlite3_step(stmt);
        if (stepResult == SQLITE_ROW) {
            if (result.columns.empty()) {
                int colCount = sqlite3_column_count(stmt);
                for (int i = 0; i < colCount; ++i) {
                    result.columns.push_back(sqlite3_column_name(stmt, i));
                }
            }
            std::vector<std::string> row;
            for (int i = 0; i < sqlite3_column_count(stmt); ++i) {
                switch (sqlite3_column_type(stmt, i)) {
                    case SQLITE_NULL:
                        row.push_back("NULL");
                        break;
                    case SQLITE_INTEGER:
                        row.push_back(std::to_string(sqlite3_column_int64(stmt, i)));
                        break;
                    case SQLITE_FLOAT:
                        row.push_back(std::to_string(sqlite3_column_double(stmt, i)));
                        break;
                    case SQLITE_TEXT:
                        row.push_back((const char*)sqlite3_column_text(stmt, i));
                        break;
                    case SQLITE_BLOB:
                        row.push_back("[BLOB]");
                        break;
                }
            }
            result.rows.push_back(row);
        } else if (stepResult == SQLITE_DONE) {
            result.affectedRows = sqlite3_changes(m_db);
            break;
        } else {
            result.success = false;
            result.error = sqlite3_errmsg(m_db);
            break;
        }
    }

    sqlite3_finalize(stmt);

    auto end = std::chrono::high_resolution_clock::now();
    result.executionTime = std::chrono::duration_cast<std::chrono::milliseconds>(end - start).count();

    return result;
}

std::string SqliteManager::explain(const std::string& sql) {
    if (!m_db) return "";

    std::string explainSql = "EXPLAIN QUERY PLAN " + sql;
    QueryResult result = execute(explainSql);

    std::string output;
    for (const auto& row : result.rows) {
        for (const auto& cell : row) {
            output += cell + " ";
        }
        output += "\n";
    }
    return output;
}

bool SqliteManager::beginTransaction() {
    if (!m_db || m_inTransaction) return false;
    execute("BEGIN TRANSACTION");
    m_inTransaction = true;
    return true;
}

bool SqliteManager::commit() {
    if (!m_db || !m_inTransaction) return false;
    execute("COMMIT");
    m_inTransaction = false;
    return true;
}

bool SqliteManager::rollback() {
    if (!m_db || !m_inTransaction) return false;
    execute("ROLLBACK");
    m_inTransaction = false;
    return true;
}
```

```cpp
    SqliteManager* g_manager = nullptr;

napi_value OpenDatabase(napi_env env, napi_callback_info info) {
    size_t argc = 1;
    napi_value args[1];
    napi_get_cb_info(env, info, &argc, args, nullptr, nullptr);

    char dbPath[512];
    size_t pathLen;
    napi_get_value_string_utf8(env, args[0], dbPath, sizeof(dbPath), &pathLen);

    bool success = g_manager->openDatabase(dbPath);

    napi_value result;
    napi_create_object(env, &result);
    napi_value jsSuccess;
    napi_get_boolean(env, success, &jsSuccess);
    napi_set_named_property(env, result, "success", jsSuccess);

    return result;
}

napi_value ExecuteSql(napi_env env, napi_callback_info info) {
    size_t argc = 2;
    napi_value args[2];
    napi_get_cb_info(env, info, &argc, args, nullptr, nullptr);

    char sql[4096];
    size_t sqlLen;
    napi_get_value_string_utf8(env, args[0], sql, sizeof(sql), &sqlLen);

    std::vector<std::string> bindArgs;
    bool hasArgs = false;
    napi_has_named_property(env, args[1], "length", &hasArgs);

    if (hasArgs) {
        napi_value jsArgs;
        napi_get_named_property(env, args[1], "length", nullptr);
        // ... parse array arguments
    }

    QueryResult qr = g_manager->execute(sql, bindArgs);

    napi_value result;
    napi_create_object(env, &result);

    napi_value jsColumns;
    napi_create_array(env, &jsColumns);
    for (size_t i = 0; i < qr.columns.size(); ++i) {
        napi_value col;
        napi_create_string_utf8(env, qr.columns[i].c_str(), NAPI_AUTO_LENGTH, &col);
        napi_set_element(env, jsColumns, i, col);
    }
    napi_set_named_property(env, result, "columns", jsColumns);

    napi_value jsRows;
    napi_create_array(env, &jsRows);
    for (size_t i = 0; i < qr.rows.size(); ++i) {
        napi_value jsRow;
        napi_create_array(env, &jsRow);
        for (size_t j = 0; j < qr.rows[i].size(); ++j) {
            napi_value cell;
            napi_create_string_utf8(env, qr.rows[i][j].c_str(), NAPI_AUTO_LENGTH, &cell);
            napi_set_element(env, jsRow, j, cell);
        }
        napi_set_element(env, jsRows, i, jsRow);
    }
    napi_set_named_property(env, result, "rows", jsRows);

    napi_value jsAffectedRows;
    napi_create_int32(env, qr.affectedRows, &jsAffectedRows);
    napi_set_named_property(env, result, "affectedRows", jsAffectedRows);

    napi_value jsExecutionTime;
    napi_create_int64(env, qr.executionTime, &jsExecutionTime);
    napi_set_named_property(env, result, "executionTime", jsExecutionTime);

    napi_value jsSuccess;
    napi_get_boolean(env, qr.success, &jsSuccess);
    napi_set_named_property(env, result, "success", jsSuccess);

    if (!qr.error.empty()) {
        napi_value jsError;
        napi_create_string_utf8(env, qr.error.c_str(), NAPI_AUTO_LENGTH, &jsError);
        napi_set_named_property(env, result, "error", jsError);
    }

    return result;
}
```

### 1.2 ArkTS 数据库服务 (DatabaseService)

**文件位置**: `entry/src/main/ets/services/DatabaseService.ets`

**模块职责**: 封装数据库操作，为 UI 层提供统一的异步接口

```typescript
// services/DatabaseService.ets
import { native } from '@kit.NAPILibrary';
import { common } from '@kit.AbilityKit';
import { fileIo } from '@kit.CoreFileKit';

interface DatabaseConfig {
  name: string;
  path: string;
}

interface Database {
  path: string;
  isOpen: boolean;
}

interface ColumnInfo {
  name: string;
  dataType: string;
  notNull: boolean;
  primaryKey: boolean;
  defaultValue?: string;
}

interface TableInfo {
  name: string;
  type: 'table' | 'view';
  rowCount: number;
  columns: ColumnInfo[];
}

interface IndexInfo {
  name: string;
  tableName: string;
  unique: boolean;
  columns: string[];
}

interface QueryResult {
  success: boolean;
  columns: string[];
  rows: Record<string, ValueType>[];
  affectedRows: number;
  executionTime: number;
  error?: string;
}

class DatabaseService {
  private currentDb: Database | null = null;
  private nativeManager: any = null;

  constructor() {
    // Initialize native module
    this.nativeManager = native.sqlite;
  }

  /**
   * Create a new database file
   */
  async createDatabase(name: string, path?: string): Promise<Database> {
    const ctx = AppStorage.get<common.UIAbilityContext>('abilityContext');
    if (!ctx) {
      throw new Error('Unable to get ability context');
    }

    // Determine storage path
    const dbPath = path || `${ctx.filesDir}/${name}.db`;

    // Create empty file
    const file = fileIo.openSync(dbPath, fileIo.OpenMode.CREATE);
    fileIo.closeSync(file);

    // Open the newly created database
    return await this.openDatabase(dbPath);
  }

  /**
   * Open an existing database file
   */
  async openDatabase(dbPath: string): Promise<Database> {
    if (this.currentDb && this.currentDb.isOpen) {
      await this.closeDatabase();
    }

    const result = await this.nativeManager.openDatabase(dbPath);

    if (result.success) {
      this.currentDb = {
        path: dbPath,
        isOpen: true
      };
      return this.currentDb;
    } else {
      throw new Error(`Failed to open database: ${result.error || 'Unknown error'}`);
    }
  }

  /**
   * Close the current database
   */
  async closeDatabase(): Promise<void> {
    if (this.currentDb && this.currentDb.isOpen) {
      await this.nativeManager.closeDatabase();
      this.currentDb = null;
    }
  }

  /**
   * Get list of all tables
   */
  async getTables(): Promise<TableInfo[]> {
    if (!this.currentDb || !this.currentDb.isOpen) {
      throw new Error('No database is currently open');
    }

    const tables: TableInfo[] = await this.nativeManager.getTables();

    // Fetch columns for each table
    for (let i = 0; i < tables.length; i++) {
      const table = tables[i];
      table.columns = await this.getColumns(table.name);
      table.rowCount = await this.getRowCount(table.name);
    }

    return tables;
  }

  /**
   * Get column information for a table
   */
  async getColumns(tableName: string): Promise<ColumnInfo[]> {
    if (!this.currentDb || !this.currentDb.isOpen) {
      throw new Error('No database is currently open');
    }

    return await this.nativeManager.getColumns(tableName);
  }

  /**
   * Get list of all indexes
   */
  async getIndexes(tableName?: string): Promise<IndexInfo[]> {
    if (!this.currentDb || !this.currentDb.isOpen) {
      throw new Error('No database is currently open');
    }

    return await this.nativeManager.getIndexes(tableName || '');
  }

  /**
   * Execute a raw SQL query
   */
  async executeSql(sql: string, bindArgs?: ValueType[]): Promise<QueryResult> {
    if (!this.currentDb || !this.currentDb.isOpen) {
      throw new Error('No database is currently open');
    }

    const argsArray = bindArgs ? Array.from(bindArgs) : [];
    const result = await this.nativeManager.executeSql(sql, argsArray);

    return {
      success: result.success,
      columns: result.columns || [],
      rows: result.rows || [],
      affectedRows: result.affectedRows || 0,
      executionTime: result.executionTime || 0,
      error: result.error
    };
  }

  /**
   * SELECT query helper
   */
  async select(
    table: string,
    columns?: string[],
    where?: string,
    orderBy?: string,
    limit?: number,
    offset?: number
  ): Promise<QueryResult> {
    let sql = 'SELECT ';

    if (columns && columns.length > 0) {
      sql += columns.join(', ');
    } else {
      sql += '*';
    }

    sql += ` FROM ${table}`;

    if (where) {
      sql += ` WHERE ${where}`;
    }

    if (orderBy) {
      sql += ` ORDER BY ${orderBy}`;
    }

    if (limit !== undefined) {
      sql += ` LIMIT ${limit}`;
      if (offset !== undefined) {
        sql += ` OFFSET ${offset}`;
      }
    }

    return await this.executeSql(sql);
  }

  /**
   * INSERT helper
   */
  async insert(table: string, data: Record<string, ValueType>): Promise<number> {
    const columns = Object.keys(data);
    const values = Object.values(data);
    const placeholders = columns.map(() => '?').join(', ');

    const sql = `INSERT INTO ${table} (${columns.join(', ')}) VALUES (${placeholders})`;

    const result = await this.executeSql(sql, values);
    return result.affectedRows;
  }

  /**
   * UPDATE helper
   */
  async update(
    table: string,
    data: Record<string, ValueType>,
    where: string
  ): Promise<number> {
    const sets = Object.keys(data).map(k => `${k} = ?`).join(', ');
    const values = [...Object.values(data)];

    const sql = `UPDATE ${table} SET ${sets} WHERE ${where}`;

    const result = await this.executeSql(sql, values);
    return result.affectedRows;
  }

  /**
   * DELETE helper
   */
  async delete(table: string, where: string): Promise<number> {
    const sql = `DELETE FROM ${table} WHERE ${where}`;
    const result = await this.executeSql(sql);
    return result.affectedRows;
  }

  /**
   * Get row count for a table
   */
  async getRowCount(tableName: string): Promise<number> {
    const result = await this.executeSql(`SELECT COUNT(*) as count FROM ${tableName}`);
    if (result.success && result.rows.length > 0) {
      return parseInt(result.rows[0]['count'] || '0', 10);
    }
    return 0;
  }

  /**
   * Begin a transaction
   */
  async beginTransaction(): Promise<void> {
    if (!this.currentDb || !this.currentDb.isOpen) {
      throw new Error('No database is currently open');
    }
    await this.nativeManager.beginTransaction();
  }

  /**
   * Commit a transaction
   */
  async commit(): Promise<void> {
    if (!this.currentDb || !this.currentDb.isOpen) {
      throw new Error('No database is currently open');
    }
    await this.nativeManager.commitTransaction();
  }

  /**
   * Rollback a transaction
   */
  async rollback(): Promise<void> {
    if (!this.currentDb || !this.currentDb.isOpen) {
      throw new Error('No database is currently open');
    }
    await this.nativeManager.rollbackTransaction();
  }

  /**
   * Get current database info
   */
  getCurrentDatabase(): Database | null {
    return this.currentDb;
  }
}
```

**文件位置**: `entry/src/main/ets/services/FileTransferService.ets`

**模块职责**: 处理文件在应用沙箱与外部存储之间的拷贝，解决 HarmonyOS 沙箱隔离问题

```typescript
// services/FileTransferService.ets
import { common } from '@kit.AbilityKit';
import { fileIo } from '@kit.CoreFileKit';
import { picker } from '@kit.CoreFileKit';

class FileTransferService {
  private context: common.UIAbilityContext;
  private sandboxDbDir: string = '';

  async init(context: common.UIAbilityContext): Promise<void> {
    this.context = context;
    // 获取沙箱内数据库存储目录
    this.sandboxDbDir = this.context.databaseDir + '/databases';
    // 确保目录存在
    if (!fileIo.accessSync(this.sandboxDbDir)) {
      fileIo.mkdirSync(this.sandboxDbDir);
    }
  }

  /**
   * 获取沙箱数据库路径
   */
  getSandboxDatabasePath(): string {
    return this.sandboxDbDir;
  }

  /**
   * 通过系统文件选择器选择并拷贝数据库到沙箱
   * @returns 沙箱内的新路径
   */
  async selectAndCopyToSandbox(): Promise<string> {
    // 1. 打开系统文件选择器
    const documentPicker = new picker.DocumentViewPicker();
    const results = await documentPicker.select({
      fileSuffixFilters: ['.db', '.sqlite', '.sqlite3']
    });

    if (results.length === 0) {
      throw new Error('用户取消了文件选择');
    }

    const sourcePath = results[0];

    // 2. 拷贝到沙箱
    return await this.copyToSandbox(sourcePath);
  }

  /**
   * 将外部数据库文件拷贝到沙箱
   */
  async copyToSandbox(sourcePath: string): Promise<string> {
    const fileName = sourcePath.split('/').pop() || 'database.db';
    const destPath = `${this.sandboxDbDir}/${Date.now()}_${fileName}`;

    // 读取源文件
    const sourceFile = fileIo.openSync(sourcePath, fileIo.OpenMode.READ_ONLY);
    const sourceStats = fileIo.statSync(sourceFile.fd);

    // 创建目标文件
    const destFile = fileIo.openSync(destPath, fileIo.OpenMode.CREATE | fileIo.OpenMode.WRITE_ONLY);

    // 拷贝内容
    const bufferSize = 64 * 1024; // 64KB buffer
    let offset = 0;
    while (offset < sourceStats.size) {
      const toRead = Math.min(bufferSize, sourceStats.size - offset);
      const buffer = new ArrayBuffer(toRead);
      fileIo.readSync(sourceFile.fd, buffer, { offset });
      fileIo.writeSync(destFile.fd, buffer, { offset });
      offset += toRead;
    }

    fileIo.closeSync(sourceFile);
    fileIo.closeSync(destFile);

    return destPath;
  }

  /**
   * 将沙箱内的数据库导出到外部位置
   */
  async exportToOriginal(sandboxPath: string, destPath?: string): Promise<void> {
    let targetPath = destPath;

    if (!targetPath) {
      // 打开保存对话框
      const documentSaver = new picker.DocumentViewPicker();
      const results = await documentSaver.save({
        suggestedName: sandboxPath.split('/').pop() || 'export.db'
      });

      if (results.length === 0) {
        throw new Error('用户取消了保存');
      }
      targetPath = results[0];
    }

    // 拷贝文件
    const sourceFile = fileIo.openSync(sandboxPath, fileIo.OpenMode.READ_ONLY);
    const destFile = fileIo.openSync(targetPath, fileIo.OpenMode.CREATE | fileIo.OpenMode.WRITE_ONLY);

    const sourceStats = fileIo.statSync(sourceFile.fd);
    const bufferSize = 64 * 1024;
    let offset = 0;

    while (offset < sourceStats.size) {
      const toRead = Math.min(bufferSize, sourceStats.size - offset);
      const buffer = new ArrayBuffer(toRead);
      fileIo.readSync(sourceFile.fd, buffer, { offset });
      fileIo.writeSync(destFile.fd, buffer, { offset });
      offset += toRead;
    }

    fileIo.closeSync(sourceFile);
    fileIo.closeSync(destFile);
  }

  /**
   * 删除沙箱内的数据库副本
   */
  async deleteSandboxCopy(sandboxPath: string): Promise<void> {
    if (fileIo.accessSync(sandboxPath)) {
      fileIo.unlinkSync(sandboxPath);
    }
  }

  /**
   * 检查沙箱内是否存在指定文件
   */
  existsInSandbox(fileName: string): boolean {
    const fullPath = `${this.sandboxDbDir}/${fileName}`;
    return fileIo.accessSync(fullPath);
  }
}

export default new FileTransferService();
```

```typescript
   */
  async getRecentDatabases(): Promise<RecentDatabase[]> {
    if (!this.prefs) await this.init(this.context);

    const json = await this.prefs.get('recent_databases', '[]');
    return JSON.parse(json) as RecentDatabase[];
  }

  /**
   * 添加到最近使用列表
   */
  async addRecentDatabase(db: RecentDatabase): Promise<void> {
    if (!this.prefs) await this.init(this.context);

    const recentList = await this.getRecentDatabases();

    // 移除已存在的同名记录
    const filtered = recentList.filter(r => r.sandboxPath !== db.sandboxPath);

    // 添加到列表头部
    filtered.unshift(db);

    // 只保留最近 10 条
    const trimmed = filtered.slice(0, 10);

    await this.prefs.set('recent_databases', JSON.stringify(trimmed));
    await this.prefs.flush();
  }

  /**
   * 清除最近使用列表
   */
  async clearRecentDatabases(): Promise<void> {
    if (!this.prefs) await this.init(this.context);
    await this.prefs.delete('recent_databases');
    await this.prefs.flush();
  }

  /**
   * 获取用户偏好设置
   */
  async getPreference(key: string, defaultValue: string): Promise<string> {
    if (!this.prefs) await this.init(this.context);
    return await this.prefs.get(key, defaultValue);
  }

  /**
   * 设置用户偏好
   */
  async setPreference(key: string, value: string): Promise<void> {
    if (!this.prefs) await this.init(this.context);
    await this.prefs.set(key, value);
    await this.prefs.flush();
  }
}

export default new PreferencesService();
```

| lastOpenedDb | string | 上次打开的数据库路径 |
| editor_fontSize | number | 编辑器字体大小 |
| window_width | number | 窗口宽度 |
| window_height | number | 窗口高度 |
